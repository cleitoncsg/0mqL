58 void zmq::reaper_t::in_event ()
59 {
60  while (true) {
61 
62  // Get the next command. If there is none, exit.
63  command_t cmd;
64  int rc = mailbox.recv (&cmd, 0);
65  if (rc != 0 && errno == EINTR)
66  continue;
67  if (rc != 0 && errno == EAGAIN)
68  break;
69  errno_assert (rc == 0);
70 
71  // Process the command.
72  cmd.destination->process_command (cmd);
73  }
74 }

86 void zmq::reaper_t::process_stop ()
87 {
88  terminating = true;
89 
90  // If there are no sockets being reaped finish immediately.
91  if (!sockets) {
92  send_done ();
93  poller->rm_fd (mailbox_handle);
94  poller->stop ();
95  }
96 }

98 void zmq::reaper_t::process_reap (socket_base_t *socket_)
99 {

106 void zmq::reaper_t::process_reaped ()
107 {
108  --sockets;
109 
110  // If reaped was already asked to terminate and there are no more sockets,
111  // finish immediately.
112  if (!sockets && terminating) {
113  send_done ();
114  poller->rm_fd (mailbox_handle);
115  poller->stop ();
116  }
