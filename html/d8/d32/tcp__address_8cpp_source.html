54 // On Solaris platform, network interface name can be queried by ioctl.
55 int zmq::tcp_address_t::resolve_nic_name (const char *nic_, bool ipv4only_)
56 {
57  // TODO: Unused parameter, IPv6 support not implemented for Solaris.
58  (void) ipv4only_;
59 
60  // Create a socket.
61  int fd = open_socket (AF_INET, SOCK_DGRAM, 0);
62  errno_assert (fd != -1);
63 
64  // Retrieve number of interfaces.
65  lifnum ifn;
66  ifn.lifn_family = AF_INET;
67  ifn.lifn_flags = 0;
68  int rc = ioctl (fd, SIOCGLIFNUM, (char*) &ifn);
69  errno_assert (rc != -1);
70 
71  // Allocate memory to get interface names.
72  size_t ifr_size = sizeof (struct lifreq) * ifn.lifn_count;
73  char *ifr = (char*) malloc (ifr_size);
74  alloc_assert (ifr);
75 
76  // Retrieve interface names.
77  lifconf ifc;
78  ifc.lifc_family = AF_INET;
79  ifc.lifc_flags = 0;
80  ifc.lifc_len = ifr_size;
81  ifc.lifc_buf = ifr;
82  rc = ioctl (fd, SIOCGLIFCONF, (char*) &ifc);
83  errno_assert (rc != -1);
84 
85  // Find the interface with the specified name and AF_INET family.
86  bool found = false;
87  lifreq *ifrp = ifc.lifc_req;
88  for (int n = 0; n < (int) (ifc.lifc_len / sizeof (lifreq));
89  n ++, ifrp ++) {
90  if (!strcmp (nic_, ifrp->lifr_name)) {
91  rc = ioctl (fd, SIOCGLIFADDR, (char*) ifrp);
92  errno_assert (rc != -1);
93  if (ifrp->lifr_addr.ss_family == AF_INET) {
94  address.ipv4 = *(sockaddr_in*) &ifrp->lifr_addr;
95  found = true;
96  break;
97  }
98  }
99  }
100 
101  // Clean-up.
102  free (ifr);
103  close (fd);
104 
105  if (!found) {
106  errno = ENODEV;
107  return -1;
108  }
109 
110  return 0;
111 }

160 int zmq::tcp_address_t::resolve_nic_name (const char *nic_, bool ipv4only_)
161 {
162  // Get the addresses.
163  ifaddrs* ifa = NULL;
164  int rc = getifaddrs (&ifa);
165  errno_assert (rc == 0);
166  zmq_assert (ifa != NULL);
167 
168  // Find the corresponding network interface.
169  bool found = false;
170  for (ifaddrs *ifp = ifa; ifp != NULL ;ifp = ifp->ifa_next)
171  {
172  if (ifp->ifa_addr == NULL)
173  continue;
174 
175  int family = ifp->ifa_addr->sa_family;
176 
177  if ((family == AF_INET
178  || (!ipv4only_ && family == AF_INET6))
179  && !strcmp (nic_, ifp->ifa_name))
180  {
181  memcpy (&address, ifp->ifa_addr,
182  (family == AF_INET) ? sizeof (struct sockaddr_in)
183  : sizeof (struct sockaddr_in6));
184  found = true;
185  break;
186  }
187  }
188 
189  // Clean-up;
190  freeifaddrs (ifa);
191 
192  if (!found) {
193  errno = ENODEV;
194  return -1;
195  }
196 
197  return 0;
198 }

216 int zmq::tcp_address_t::resolve_interface (const char *interface_,
217  bool ipv4only_)
218 {
219  // Initialize temporary output pointers with storage address.
220  sockaddr_storage ss;
221  sockaddr *out_addr = (sockaddr*) &ss;
222  size_t out_addrlen;
223 
224  // Initialise IP-format family/port and populate temporary output pointers
225  // with the address.
226  if (ipv4only_) {
227  sockaddr_in ip4_addr;
228  memset (&ip4_addr, 0, sizeof (ip4_addr));
229  ip4_addr.sin_family = AF_INET;
230  ip4_addr.sin_addr.s_addr = htonl (INADDR_ANY);
231  out_addrlen = sizeof ip4_addr;
232  memcpy (out_addr, &ip4_addr, out_addrlen);
233  }
234  else {
235  sockaddr_in6 ip6_addr;
236  memset (&ip6_addr, 0, sizeof (ip6_addr));
237  ip6_addr.sin6_family = AF_INET6;
238  memcpy (&ip6_addr.sin6_addr, &in6addr_any, sizeof (in6addr_any));
239  out_addrlen = sizeof ip6_addr;
240  memcpy (out_addr, &ip6_addr, out_addrlen);
241  }
242 
243  // * resolves to INADDR_ANY or in6addr_any.
244  if (strcmp (interface_, "*") == 0) {
245  zmq_assert (out_addrlen <= sizeof address);
246  memcpy (&address, out_addr, out_addrlen);
247  return 0;
248  }
249 
250  // Try to resolve the string as a NIC name.
251  int rc = resolve_nic_name (interface_, ipv4only_);
252  if (rc != 0 && errno != ENODEV)
253  return rc;
254  if (rc == 0)
255  return 0;
256 
257  // There's no such interface name. Assume literal address.
258 #if defined ZMQ_HAVE_OPENVMS && defined __ia64
259  __addrinfo64 *res = NULL;
260  __addrinfo64 req;
261 #else
262  addrinfo *res = NULL;
263  addrinfo req;
264 #endif
265  memset (&req, 0, sizeof (req));
266 
267  // Choose IPv4 or IPv6 protocol family. Note that IPv6 allows for
268  // IPv4-in-IPv6 addresses.
269  req.ai_family = ipv4only_ ? AF_INET : AF_INET6;
270 
271  // Arbitrary, not used in the output, but avoids duplicate results.
272  req.ai_socktype = SOCK_STREAM;
273 
274  // Restrict hostname/service to literals to avoid any DNS lookups or
275  // service-name irregularity due to indeterminate socktype.
276  req.ai_flags = AI_PASSIVE | AI_NUMERICHOST;
277 
278 #if defined AI_V4MAPPED && !defined ZMQ_HAVE_FREEBSD
279  // In this API we only require IPv4-mapped addresses when
280  // no native IPv6 interfaces are available (~AI_ALL).
281  // This saves an additional DNS roundtrip for IPv4 addresses.
282  // Note: While the AI_V4MAPPED flag is defined on FreeBSD system,
283  // it is not supported here. See libzmq issue #331.
284  if (req.ai_family == AF_INET6)
285  req.ai_flags |= AI_V4MAPPED;
286 #endif
287 
288  // Resolve the literal address. Some of the error info is lost in case
289  // of error, however, there's no way to report EAI errors via errno.
290  rc = getaddrinfo (interface_, NULL, &req, &res);
291  if (rc) {
292  errno = ENODEV;
293  return -1;
294  }
295 
296  // Use the first result.
297  zmq_assert (res != NULL);
298  zmq_assert ((size_t) (res->ai_addrlen) <= sizeof (address));
299  memcpy (&address, res->ai_addr, res->ai_addrlen);
300 
301  // Cleanup getaddrinfo after copying the possibly referenced result.
302  freeaddrinfo (res);
303 
304  return 0;
305 }

307 int zmq::tcp_address_t::resolve_hostname (const char *hostname_, bool ipv4only_)
308 {
309  // Set up the query.
310 #if defined ZMQ_HAVE_OPENVMS && defined __ia64 && __INITIAL_POINTER_SIZE == 64
311  __addrinfo64 req;
312 #else
313  addrinfo req;
314 #endif
315  memset (&req, 0, sizeof (req));
316 
317  // Choose IPv4 or IPv6 protocol family. Note that IPv6 allows for
318  // IPv4-in-IPv6 addresses.
319  req.ai_family = ipv4only_ ? AF_INET : AF_INET6;
320 
321  // Need to choose one to avoid duplicate results from getaddrinfo() - this
322  // doesn't really matter, since it's not included in the addr-output.
323  req.ai_socktype = SOCK_STREAM;
324 
325 #if defined AI_V4MAPPED && !defined ZMQ_HAVE_FREEBSD
326  // In this API we only require IPv4-mapped addresses when
327  // no native IPv6 interfaces are available.
328  // This saves an additional DNS roundtrip for IPv4 addresses.
329  // Note: While the AI_V4MAPPED flag is defined on FreeBSD system,
330  // it is not supported here. See libzmq issue #331.
331  if (req.ai_family == AF_INET6)
332  req.ai_flags |= AI_V4MAPPED;
333 #endif
334 
335  // Resolve host name. Some of the error info is lost in case of error,
336  // however, there's no way to report EAI errors via errno.
337 #if defined ZMQ_HAVE_OPENVMS && defined __ia64 && __INITIAL_POINTER_SIZE == 64
338  __addrinfo64 *res;
339 #else
340  addrinfo *res;
341 #endif
342  int rc = getaddrinfo (hostname_, NULL, &req, &res);
343  if (rc) {
344  switch (rc) {
345  case EAI_MEMORY:
346  errno = ENOMEM;
347  break;
348  default:
349  errno = EINVAL;
350  break;
351  }
352  return -1;
353  }
354 
355  // Copy first result to output addr with hostname and service.
356  zmq_assert ((size_t) (res->ai_addrlen) <= sizeof (address));
357  memcpy (&address, res->ai_addr, res->ai_addrlen);
358 
359  freeaddrinfo (res);
360 
361  return 0;
362 }

385 int zmq::tcp_address_t::resolve (const char *name_, bool local_, bool ipv4only_)
386 {
387  // Find the ':' at end that separates address from the port number.
388  const char *delimiter = strrchr (name_, ':');
389  if (!delimiter) {
390  errno = EINVAL;
391  return -1;
392  }
393 
394  // Separate the address/port.
395  std::string addr_str (name_, delimiter - name_);
396  std::string port_str (delimiter + 1);
397 
398  // Remove square brackets around the address, if any.
399  if (addr_str.size () >= 2 && addr_str [0] == '[' &&
400  addr_str [addr_str.size () - 1] == ']')
401  addr_str = addr_str.substr (1, addr_str.size () - 2);
402 
403  uint16_t port;
404  // Allow 0 specifically, to detect invalid port error in atoi if not
405  if (port_str == "*" || port_str == "0")
406  // Resolve wildcard to 0 to allow autoselection of port
407  port = 0;
408  else {
409  // Parse the port number (0 is not a valid port).
410  port = (uint16_t) atoi (port_str.c_str ());
411  if (port == 0) {
412  errno = EINVAL;
413  return -1;
414  }
415  }
416 
417  // Resolve the IP address.
418  int rc;
419  if (local_)
420  rc = resolve_interface (addr_str.c_str (), ipv4only_);
421  else
422  rc = resolve_hostname (addr_str.c_str (), ipv4only_);
423  if (rc != 0)
424  return -1;
425 
426  // Set the port into the address structure.
427  if (address.generic.sa_family == AF_INET6)
428  address.ipv6.sin6_port = htons (port);
429  else
430  address.ipv4.sin_port = htons (port);
431 
432  return 0;
433 }

436 {
437  if (address.generic.sa_family != AF_INET && address.generic.sa_family != AF_INET6) {
438  addr_.clear ();
439  return -1;

455  else {
456  std::stringstream s;

496 int zmq::tcp_address_mask_t::resolve (const char *name_, bool ipv4only_)
497 {
498  // Find '/' at the end that separates address from the cidr mask number.
499  // Allow empty mask clause and threat it like '/32' for ipv4 or '/128' for ipv6.
500  std::string addr_str, mask_str;
501  const char *delimiter = strrchr (name_, '/');
502  if (delimiter != NULL) {
503  addr_str.assign (name_, delimiter - name_);
504  mask_str.assign (delimiter + 1);
505  if (mask_str.empty ()) {
506  errno = EINVAL;
507  return -1;
508  }
509  }
510  else {
511  addr_str.assign (name_);
512  }
513 
514  // Parse address part using standard routines.
515  int rc = tcp_address_t::resolve_hostname (addr_str.c_str (), ipv4only_);
516  if (rc != 0)
517  return rc;
518 
519  // Parse the cidr mask number.
520  if (mask_str.empty ()) {
521  if (address.generic.sa_family == AF_INET6)
522  address_mask = 128;
523  else
524  address_mask = 32;
525  }
526  else
527  if (mask_str == "0") {
528  address_mask = 0;
529  }
530  else {
531  int mask = atoi (mask_str.c_str ());
532  if (
533  (mask < 1) ||
534  (address.generic.sa_family == AF_INET6 && mask > 128) ||
535  (address.generic.sa_family != AF_INET6 && mask > 32)
536  ) {
537  errno = EINVAL;
538  return -1;
539  }
540  address_mask = mask;
541  }
542 
543  return 0;
544 }

577 bool zmq::tcp_address_mask_t::match_address (const struct sockaddr *ss, const socklen_t ss_len) const
578 {
579  zmq_assert (address_mask != -1 && ss != NULL && ss_len >= (socklen_t) sizeof (struct sockaddr));
580 
581  if (ss->sa_family != address.generic.sa_family)
582  return false;
583 
584  if (address_mask > 0) {
585  int mask;
586  const uint8_t *our_bytes, *their_bytes;
587  if (ss->sa_family == AF_INET6) {
588  zmq_assert (ss_len == sizeof (struct sockaddr_in6));
589  their_bytes = (const uint8_t *) &(((const struct sockaddr_in6 *) ss)->sin6_addr);
590  our_bytes = (const uint8_t *) &address.ipv6.sin6_addr;
591  mask = sizeof (struct in6_addr) * 8;
592  }
593  else {
594  zmq_assert (ss_len == sizeof (struct sockaddr_in));
595  their_bytes = (const uint8_t *) &(((const struct sockaddr_in *) ss)->sin_addr);
596  our_bytes = (const uint8_t *) &address.ipv4.sin_addr;
597  mask = sizeof (struct in_addr) * 8;
598  }
599  if (address_mask < mask) mask = address_mask;
600 
601  size_t full_bytes = mask / 8;
602  if (memcmp(our_bytes, their_bytes, full_bytes))
603  return false;
604 
605  uint8_t last_byte_bits = (0xffU << (8 - (mask % 8))) & 0xffU;
606  if (last_byte_bits) {
607  if ((their_bytes[full_bytes] & last_byte_bits) != (our_bytes[full_bytes] & last_byte_bits))
608  return false;
609  }
610  }
611 
612  return true;
613 }
