57 zmq::ctx_t::~ctx_t ()
58 {
59  // Check that there are no remaining sockets.
60  zmq_assert (sockets.empty ());
61 
62  // Ask I/O threads to terminate. If stop signal wasn't sent to I/O
63  // thread subsequent invocation of destructor would hang-up.
64  for (io_threads_t::size_type i = 0; i != io_threads.size (); i++)
65  io_threads [i]->stop ();
66 
67  // Wait till I/O threads actually terminate.
68  for (io_threads_t::size_type i = 0; i != io_threads.size (); i++)
69  delete io_threads [i];
70 
71  // Deallocate the reaper thread object.
72  if (reaper)
73  delete reaper;
74 
75  // Deallocate the array of mailboxes. No special work is
76  // needed as mailboxes themselves were deallocated with their
77  // corresponding io_thread/socket objects.
78  if (slots)
79  free (slots);
80 
81  // Remove the tag, so that the object is considered dead.
82  tag = 0xdeadbeef;
83 }

85 int zmq::ctx_t::terminate ()
86 {
87  slot_sync.lock ();
88  if (!starting) {
89 
90  // Check whether termination was already underway, but interrupted and now
91  // restarted.
92  bool restarted = terminating;
93  terminating = true;
94  slot_sync.unlock ();
95 
96  // First attempt to terminate the context.
97  if (!restarted) {
98 
99  // First send stop command to sockets so that any blocking calls
100  // can be interrupted. If there are no sockets we can ask reaper
101  // thread to stop.
102  slot_sync.lock ();
103  for (sockets_t::size_type i = 0; i != sockets.size (); i++)
104  sockets [i]->stop ();
105  if (sockets.empty ())
106  reaper->stop ();
107  slot_sync.unlock ();
108  }
109 
110  // Wait till reaper thread closes all the sockets.
111  command_t cmd;
112  int rc = term_mailbox.recv (&cmd, -1);
113  if (rc == -1 && errno == EINTR)
114  return -1;
115  errno_assert (rc == 0);
116  zmq_assert (cmd.type == command_t::done);
117  slot_sync.lock ();
118  zmq_assert (sockets.empty ());
119  }
120  slot_sync.unlock ();
121 
122  // Deallocate the resources.
123  delete this;
124 
125  return 0;
126 }

164 zmq::socket_base_t *zmq::ctx_t::create_socket (int type_)
165 {
166  slot_sync.lock ();
167  if (unlikely (starting)) {
168 
169  starting = false;
170  // Initialise the array of mailboxes. Additional three slots are for
171  // zmq_term thread and reaper thread.
172  opt_sync.lock ();
173  int mazmq = max_sockets;
174  int ios = io_thread_count;
175  opt_sync.unlock ();
176  slot_count = mazmq + ios + 2;
177  slots = (mailbox_t**) malloc (sizeof (mailbox_t*) * slot_count);
178  alloc_assert (slots);
179 
180  // Initialise the infrastructure for zmq_term thread.
181  slots [term_tid] = &term_mailbox;
182 
183  // Create the reaper thread.
184  reaper = new (std::nothrow) reaper_t (this, reaper_tid);
185  alloc_assert (reaper);
186  slots [reaper_tid] = reaper->get_mailbox ();
187  reaper->start ();
188 
189  // Create I/O thread objects and launch them.
190  for (int i = 2; i != ios + 2; i++) {
191  io_thread_t *io_thread = new (std::nothrow) io_thread_t (this, i);
192  alloc_assert (io_thread);
193  io_threads.push_back (io_thread);
194  slots [i] = io_thread->get_mailbox ();
195  io_thread->start ();
196  }
197 
198  // In the unused part of the slot array, create a list of empty slots.
199  for (int32_t i = (int32_t) slot_count - 1;
200  i >= (int32_t) ios + 2; i--) {
201  empty_slots.push_back (i);
202  slots [i] = NULL;
203  }
204  }
205 
206  // Once zmq_term() was called, we can't create new sockets.
207  if (terminating) {
208  slot_sync.unlock ();
209  errno = ETERM;
210  return NULL;
211  }
212 
213  // If max_sockets limit was reached, return error.
214  if (empty_slots.empty ()) {
215  slot_sync.unlock ();
216  errno = EMFILE;
217  return NULL;
218  }
219 
220  // Choose a slot for the socket.
221  uint32_t slot = empty_slots.back ();
222  empty_slots.pop_back ();
223 
224  // Generate new unique socket ID.
225  int sid = ((int) max_socket_id.add (1)) + 1;
226 
227  // Create the socket and register its mailbox.
228  socket_base_t *s = socket_base_t::create (type_, this, slot, sid);
229  if (!s) {
230  empty_slots.push_back (slot);
231  slot_sync.unlock ();
232  return NULL;
233  }
234  sockets.push_back (s);
235  slots [slot] = s->get_mailbox ();
236 
237  slot_sync.unlock ();
238  return s;
239 }

241 void zmq::ctx_t::destroy_socket (class socket_base_t *socket_)
242 {
243  slot_sync.lock ();
244 
245  // Free the associated thread slot.
246  uint32_t tid = socket_->get_tid ();
247  empty_slots.push_back (tid);
248  slots [tid] = NULL;
249 
250  // Remove the socket from the list of sockets.
251  sockets.erase (socket_);
252 
253  // If zmq_term() was already called and there are no more socket
254  // we can ask reaper thread to terminate.
255  if (terminating && sockets.empty ())
256  reaper->stop ();
257 
258  slot_sync.unlock ();
259 }

271 zmq::io_thread_t *zmq::ctx_t::choose_io_thread (uint64_t affinity_)
272 {
273  if (io_threads.empty ())
274  return NULL;
275 
276  // Find the I/O thread with minimum load.
277  int min_load = -1;
278  io_thread_t *selected_io_thread = NULL;
279  for (io_threads_t::size_type i = 0; i != io_threads.size (); i++) {
280  if (!affinity_ || (affinity_ & (uint64_t (1) << i))) {
281  int load = io_threads [i]->get_load ();
282  if (selected_io_thread == NULL || load < min_load) {
283  min_load = load;
284  selected_io_thread = io_threads [i];
285  }
286  }
287  }
288  return selected_io_thread;
289 }

308 void zmq::ctx_t::unregister_endpoints (socket_base_t *socket_)
309 {
310  endpoints_sync.lock ();
311 
312  endpoints_t::iterator it = endpoints.begin ();
313  while (it != endpoints.end ()) {
314  if (it->second.socket == socket_) {
315  endpoints_t::iterator to_erase = it;
316  ++it;
317  endpoints.erase (to_erase);
318  continue;
319  }
320  ++it;
321  }
322 
323  endpoints_sync.unlock ();
324 }

326 zmq::endpoint_t zmq::ctx_t::find_endpoint (const char *addr_)
327 {
328  endpoints_sync.lock ();
329 
330  endpoints_t::iterator it = endpoints.find (addr_);
331  if (it == endpoints.end ()) {
332  endpoints_sync.unlock ();
333  errno = ECONNREFUSED;
334  endpoint_t empty = {NULL, options_t()};
335  return empty;
336  }
337  endpoint_t endpoint = it->second;
338 
339  // Increment the command sequence number of the peer so that it won't
340  // get deallocated until "bind" command is issued by the caller.
341  // The subsequent 'bind' has to be called with inc_seqnum parameter
342  // set to false, so that the seqnum isn't incremented twice.
343  endpoint.socket->inc_seqnum ();
344 
345  endpoints_sync.unlock ();
346  return endpoint;
347 }
