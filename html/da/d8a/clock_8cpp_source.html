55 uint64_t zmq::clock_t::now_us ()
56 {
57 #if defined ZMQ_HAVE_WINDOWS
58 
59  // Get the high resolution counter's accuracy.
60  LARGE_INTEGER ticksPerSecond;
61  QueryPerformanceFrequency (&ticksPerSecond);
62 
63  // What time is it?
64  LARGE_INTEGER tick;
65  QueryPerformanceCounter (&tick);
66 
67  // Convert the tick number into the number of seconds
68  // since the system was started.
69  double ticks_div = (double) (ticksPerSecond.QuadPart / 1000000);
70  return (uint64_t) (tick.QuadPart / ticks_div);
71 
72 #elif defined HAVE_CLOCK_GETTIME && defined CLOCK_MONOTONIC
73 
74  // Use POSIX clock_gettime function to get precise monotonic time.
75  struct timespec tv;
76  int rc = clock_gettime (CLOCK_MONOTONIC, &tv);
77  // Fix case where system has clock_gettime but CLOCK_MONOTONIC is not supported.
78  // This should be a configuration check, but I looked into it and writing an
79  // AC_FUNC_CLOCK_MONOTONIC seems beyond my powers.
80  if( rc != 0) {
81  // Use POSIX gettimeofday function to get precise time.
82  struct timeval tv;
83  int rc = gettimeofday (&tv, NULL);
84  errno_assert (rc == 0);
85  return (tv.tv_sec * (uint64_t) 1000000 + tv.tv_usec);
86  }
87  return (tv.tv_sec * (uint64_t) 1000000 + tv.tv_nsec / 1000);
88 
89 #elif defined HAVE_GETHRTIME
90 
91  return (gethrtime () / 1000);
92 
93 #else
94 
95  // Use POSIX gettimeofday function to get precise time.
96  struct timeval tv;
97  int rc = gettimeofday (&tv, NULL);
98  errno_assert (rc == 0);
99  return (tv.tv_sec * (uint64_t) 1000000 + tv.tv_usec);
100 
101 #endif
102 }

104 uint64_t zmq::clock_t::now_ms ()
105 {
106  uint64_t tsc = rdtsc ();
107 
108  // If TSC is not supported, get precise time and chop off the microseconds.
109  if (!tsc)
110  return now_us () / 1000;
111 
112  // If TSC haven't jumped back (in case of migration to a different
113  // CPU core) and if not too much time elapsed since last measurement,
114  // we can return cached time value.
115  if (likely (tsc - last_tsc <= (clock_precision / 2) && tsc >= last_tsc))
116  return last_time;
117 
118  last_tsc = tsc;
119  last_time = now_us () / 1000;
120  return last_time;
121 }

123 uint64_t zmq::clock_t::rdtsc ()
124 {
125 #if (defined _MSC_VER && (defined _M_IX86 || defined _M_X64))
126  return __rdtsc ();
127 #elif (defined __GNUC__ && (defined __i386__ || defined __x86_64__))
128  uint32_t low, high;
129  __asm__ volatile ("rdtsc" : "=a" (low), "=d" (high));
130  return (uint64_t) high << 32 | low;
131 #elif (defined __SUNPRO_CC && (__SUNPRO_CC >= 0x5100) && (defined __i386 || \
132  defined __amd64 || defined __x86_64))
133  union {
134  uint64_t u64val;
135  uint32_t u32val [2];
136  } tsc;
137  asm("rdtsc" : "=a" (tsc.u32val [0]), "=d" (tsc.u32val [1]));
138  return tsc.u64val;
139 #elif defined(__s390__)
140  uint64_t tsc;
141  asm("\tstck\t%0\n" : "=Q" (tsc) : : "cc");
142  tsc >>= 12; /* convert to microseconds just to be consistent */
143  return(tsc);
144 #else
145  return 0;
146 #endif
147 }
