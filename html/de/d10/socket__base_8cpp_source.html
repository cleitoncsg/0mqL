96  break;

98  s = new (std::nothrow) rep_t (parent_, tid_, sid_);
99  break;
100  case ZMQ_DEALER:
101  s = new (std::nothrow) dealer_t (parent_, tid_, sid_);
102  break;
103  case ZMQ_ROUTER:
104  s = new (std::nothrow) router_t (parent_, tid_, sid_);
105  break;
106  case ZMQ_PULL:
107  s = new (std::nothrow) pull_t (parent_, tid_, sid_);
108  break;
109  case ZMQ_PUSH:
110  s = new (std::nothrow) push_t (parent_, tid_, sid_);
111  break;
112  case ZMQ_XPUB:
113  s = new (std::nothrow) xpub_t (parent_, tid_, sid_);

160 int zmq::socket_base_t::parse_uri (const char *uri_,
161  std::string &protocol_, std::string &address_)
162 {
163  zmq_assert (uri_ != NULL);
164 
165  std::string uri (uri_);
166  std::string::size_type pos = uri.find ("://");
167  if (pos == std::string::npos) {
168  errno = EINVAL;
169  return -1;
170  }
171  protocol_ = uri.substr (0, pos);
172  address_ = uri.substr (pos + 3);
173 
174  if (protocol_.empty () || address_.empty ()) {
175  errno = EINVAL;
176  return -1;
177  }
178  return 0;

181 int zmq::socket_base_t::check_protocol (const std::string &protocol_)
182 {
183  // First check out whether the protcol is something we are aware of.
184  if (protocol_ != "inproc" && protocol_ != "ipc" && protocol_ != "tcp" &&
185  protocol_ != "pgm" && protocol_ != "epgm") {
186  errno = EPROTONOSUPPORT;
187  return -1;
188  }
189 
190  // If 0MQ is not compiled with OpenPGM, pgm and epgm transports
191  // are not avaialble.
192 #if !defined ZMQ_HAVE_OPENPGM
193  if (protocol_ == "pgm" || protocol_ == "epgm") {
194  errno = EPROTONOSUPPORT;
195  return -1;
196  }
197 #endif
198 
199  // IPC transport is not available on Windows and OpenVMS.
200 #if defined ZMQ_HAVE_WINDOWS || defined ZMQ_HAVE_OPENVMS
201  if (protocol_ == "ipc") {
202  // Unknown protocol.
203  errno = EPROTONOSUPPORT;
204  return -1;
205  }
206 #endif
207 
208  // Check whether socket type and transport protocol match.
209  // Specifically, multicast protocols can't be combined with
210  // bi-directional messaging patterns (socket types).
211  if ((protocol_ == "pgm" || protocol_ == "epgm") &&
212  options.type != ZMQ_PUB && options.type != ZMQ_SUB &&
213  options.type != ZMQ_XPUB && options.type != ZMQ_XSUB) {
214  errno = ENOCOMPATPROTO;
215  return -1;
216  }
217 
218  // Protocol is available.
219  return 0;
220 }

222 void zmq::socket_base_t::attach_pipe (pipe_t *pipe_, bool icanhasall_)
223 {
224  // First, register the pipe so that we can terminate it later on.
225  pipe_->set_event_sink (this);
226  pipes.push_back (pipe_);
227 
228  // Let the derived socket type know about new pipe.
229  xattach_pipe (pipe_, icanhasall_);
230 
231  // If the socket is already being closed, ask any new pipes to terminate
232  // straight away.
233  if (is_terminating ()) {
234  register_term_acks (1);
235  pipe_->terminate (false);
236  }

257 int zmq::socket_base_t::getsockopt (int option_, void *optval_,
258  size_t *optvallen_)
259 {
260  if (unlikely (ctx_terminated)) {
261  errno = ETERM;
262  return -1;
263  }
264 
265  if (option_ == ZMQ_RCVMORE) {
266  if (*optvallen_ < sizeof (int)) {
267  errno = EINVAL;
268  return -1;
269  }
270  *((int*) optval_) = rcvmore ? 1 : 0;
271  *optvallen_ = sizeof (int);
272  return 0;
273  }
274 
275  if (option_ == ZMQ_FD) {
276  if (*optvallen_ < sizeof (fd_t)) {
277  errno = EINVAL;
278  return -1;
279  }
280  *((fd_t*) optval_) = mailbox.get_fd ();
281  *optvallen_ = sizeof (fd_t);
282  return 0;
283  }
284 
285  if (option_ == ZMQ_EVENTS) {
286  if (*optvallen_ < sizeof (int)) {
287  errno = EINVAL;
288  return -1;
289  }
290  int rc = process_commands (0, false);
291  if (rc != 0 && (errno == EINTR || errno == ETERM))
292  return -1;
293  errno_assert (rc == 0);
294  *((int*) optval_) = 0;
295  if (has_out ())
296  *((int*) optval_) |= ZMQ_POLLOUT;
297  if (has_in ())
298  *((int*) optval_) |= ZMQ_POLLIN;
299  *optvallen_ = sizeof (int);
300  return 0;
301  }
302 
303  return options.getsockopt (option_, optval_, optvallen_);
304 }

306 int zmq::socket_base_t::bind (const char *addr_)
307 {
308  if (unlikely (ctx_terminated)) {
309  errno = ETERM;
310  return -1;
311  }
312 
313  // Process pending commands, if any.
314  int rc = process_commands (0, false);
315  if (unlikely (rc != 0))
316  return -1;
317 
318  // Parse addr_ string.
319  std::string protocol;
320  std::string address;
321  rc = parse_uri (addr_, protocol, address);
322  if (rc != 0)
323  return -1;
324 
325  rc = check_protocol (protocol);
326  if (rc != 0)
327  return -1;
328 
329  if (protocol == "inproc") {
330  endpoint_t endpoint = {this, options};
331  int rc = register_endpoint (addr_, endpoint);
332  if (rc == 0) {
333  // Save last endpoint URI
334  options.last_endpoint.assign (addr_);
335  }
336  return rc;
337  }
338 
339  if (protocol == "pgm" || protocol == "epgm") {
340  // For convenience's sake, bind can be used interchageable with
341  // connect for PGM and EPGM transports.
342  return connect (addr_);
343  }
344 
345  // Remaining trasnports require to be run in an I/O thread, so at this
346  // point we'll choose one.
347  io_thread_t *io_thread = choose_io_thread (options.affinity);
348  if (!io_thread) {
349  errno = EMTHREAD;
350  return -1;
351  }
352 
353  if (protocol == "tcp") {
354  tcp_listener_t *listener = new (std::nothrow) tcp_listener_t (
355  io_thread, this, options);
356  alloc_assert (listener);
357  int rc = listener->set_address (address.c_str ());
358  if (rc != 0) {
359  delete listener;
360  event_bind_failed (addr_, zmq_errno());
361  return -1;
362  }
363 
364  // Save last endpoint URI
365  listener->get_address (options.last_endpoint);
366 
367  add_endpoint (addr_, (own_t *) listener);
368  return 0;
369  }
370 
371 #if !defined ZMQ_HAVE_WINDOWS && !defined ZMQ_HAVE_OPENVMS
372  if (protocol == "ipc") {
373  ipc_listener_t *listener = new (std::nothrow) ipc_listener_t (
374  io_thread, this, options);
375  alloc_assert (listener);
376  int rc = listener->set_address (address.c_str ());
377  if (rc != 0) {
378  delete listener;
379  event_bind_failed (addr_, zmq_errno());
380  return -1;
381  }
382 
383  // Save last endpoint URI
384  listener->get_address (options.last_endpoint);
385 
386  add_endpoint (addr_, (own_t *) listener);
387  return 0;
388  }
389 #endif
390 
391  zmq_assert (false);
392  return -1;
393 }

395 int zmq::socket_base_t::connect (const char *addr_)
396 {
397  if (unlikely (ctx_terminated)) {
398  errno = ETERM;
399  return -1;
400  }
401 
402  // Process pending commands, if any.
403  int rc = process_commands (0, false);
404  if (unlikely (rc != 0))
405  return -1;
406 
407  // Parse addr_ string.
408  std::string protocol;
409  std::string address;
410  rc = parse_uri (addr_, protocol, address);
411  if (rc != 0)
412  return -1;
413 
414  rc = check_protocol (protocol);
415  if (rc != 0)
416  return -1;
417 
418  if (protocol == "inproc") {
419 
420  // TODO: inproc connect is specific with respect to creating pipes
421  // as there's no 'reconnect' functionality implemented. Once that
422  // is in place we should follow generic pipe creation algorithm.
423 
424  // Find the peer endpoint.
425  endpoint_t peer = find_endpoint (addr_);
426  if (!peer.socket)
427  return -1;
428 
429  // The total HWM for an inproc connection should be the sum of
430  // the binder's HWM and the connector's HWM.
431  int sndhwm;
432  int rcvhwm;
433  if (options.sndhwm == 0 || peer.options.rcvhwm == 0)
434  sndhwm = 0;
435  else
436  sndhwm = options.sndhwm + peer.options.rcvhwm;
437  if (options.rcvhwm == 0 || peer.options.sndhwm == 0)
438  rcvhwm = 0;
439  else
440  rcvhwm = options.rcvhwm + peer.options.sndhwm;
441 
442  // Create a bi-directional pipe to connect the peers.
443  object_t *parents [2] = {this, peer.socket};
444  pipe_t *pipes [2] = {NULL, NULL};
445  int hwms [2] = {sndhwm, rcvhwm};
446  bool delays [2] = {options.delay_on_disconnect, options.delay_on_close};
447  int rc = pipepair (parents, pipes, hwms, delays);
448  errno_assert (rc == 0);
449 
450  // Attach local end of the pipe to this socket object.
451  attach_pipe (pipes [0]);
452 
453  // If required, send the identity of the local socket to the peer.
454  if (peer.options.recv_identity) {
455  msg_t id;
456  rc = id.init_size (options.identity_size);
457  errno_assert (rc == 0);
458  memcpy (id.data (), options.identity, options.identity_size);
459  id.set_flags (msg_t::identity);
460  bool written = pipes [0]->write (&id);
461  zmq_assert (written);
462  pipes [0]->flush ();
463  }
464 
465  // If required, send the identity of the peer to the local socket.
466  if (options.recv_identity) {
467  msg_t id;
468  rc = id.init_size (peer.options.identity_size);
469  errno_assert (rc == 0);
470  memcpy (id.data (), peer.options.identity, peer.options.identity_size);
471  id.set_flags (msg_t::identity);
472  bool written = pipes [1]->write (&id);
473  zmq_assert (written);
474  pipes [1]->flush ();
475  }
476 
477  // Attach remote end of the pipe to the peer socket. Note that peer's
478  // seqnum was incremented in find_endpoint function. We don't need it
479  // increased here.
480  send_bind (peer.socket, pipes [1], false);
481 
482  // Save last endpoint URI
483  options.last_endpoint.assign (addr_);
484 
485  return 0;
486  }
487 
488  // Choose the I/O thread to run the session in.
489  io_thread_t *io_thread = choose_io_thread (options.affinity);
490  if (!io_thread) {
491  errno = EMTHREAD;
492  return -1;
493  }
494 
495  address_t *paddr = new (std::nothrow) address_t (protocol, address);
496  alloc_assert (paddr);
497 
498  // Resolve address (if needed by the protocol)
499  if (protocol == "tcp") {
500  paddr->resolved.tcp_addr = new (std::nothrow) tcp_address_t ();
501  alloc_assert (paddr->resolved.tcp_addr);
502  int rc = paddr->resolved.tcp_addr->resolve (
503  address.c_str (), false, options.ipv4only ? true : false);
504  if (rc != 0) {
505  delete paddr;
506  return -1;
507  }
508  }
509 #if !defined ZMQ_HAVE_WINDOWS && !defined ZMQ_HAVE_OPENVMS
510  else if(protocol == "ipc") {
511  paddr->resolved.ipc_addr = new (std::nothrow) ipc_address_t ();
512  alloc_assert (paddr->resolved.ipc_addr);
513  int rc = paddr->resolved.ipc_addr->resolve (address.c_str ());
514  if (rc != 0) {
515  delete paddr;
516  return -1;
517  }
518  }
519 #endif
520 #ifdef ZMQ_HAVE_OPENPGM
521  if (protocol == "pgm" || protocol == "epgm") {
522  struct pgm_addrinfo_t *res = NULL;
523  uint16_t port_number = 0;
524  int rc = pgm_socket_t::init_address(address.c_str(), &res, &port_number);
525  if (res != NULL)
526  pgm_freeaddrinfo (res);
527  if (rc != 0 || port_number == 0)
528  return -1;
529  }
530 #endif
531  // Create session.
532  session_base_t *session = session_base_t::create (io_thread, true, this,
533  options, paddr);
534  errno_assert (session);
535 
536  // PGM does not support subscription forwarding; ask for all data to be
537  // sent to this pipe.
538  bool icanhasall = protocol == "pgm" || protocol == "epgm";
539 
540  if (options.delay_attach_on_connect != 1 || icanhasall) {
541  // Create a bi-directional pipe.
542  object_t *parents [2] = {this, session};
543  pipe_t *pipes [2] = {NULL, NULL};
544  int hwms [2] = {options.sndhwm, options.rcvhwm};
545  bool delays [2] = {options.delay_on_disconnect, options.delay_on_close};
546  rc = pipepair (parents, pipes, hwms, delays);
547  errno_assert (rc == 0);
548 
549  // Attach local end of the pipe to the socket object.
550  attach_pipe (pipes [0], icanhasall);
551 
552  // Attach remote end of the pipe to the session object later on.
553  session->attach_pipe (pipes [1]);
554  }
555 
556  // Save last endpoint URI
557  paddr->to_string (options.last_endpoint);
558 
559  add_endpoint (addr_, (own_t *) session);
560  return 0;
561 }

570 int zmq::socket_base_t::term_endpoint (const char *addr_)
571 {
572  // Check whether the library haven't been shut down yet.
573  if (unlikely (ctx_terminated)) {
574  errno = ETERM;
575  return -1;
576  }
577 
578  // Check whether endpoint address passed to the function is valid.
579  if (unlikely (!addr_)) {
580  errno = EINVAL;
581  return -1;
582  }
583 
584  // Process pending commands, if any, since there could be pending unprocessed process_own()'s
585  // (from launch_child() for example) we're asked to terminate now.
586  int rc = process_commands (0, false);
587  if (unlikely (rc != 0))
588  return -1;
589 
590  // Find the endpoints range (if any) corresponding to the addr_ string.
591  std::pair <endpoints_t::iterator, endpoints_t::iterator> range = endpoints.equal_range (std::string (addr_));
592  if (range.first == range.second)
593  return -1;
594 
595  for (endpoints_t::iterator it = range.first; it != range.second; ++it)
596  term_child (it->second);
597  endpoints.erase (range.first, range.second);
598  return 0;
599 }

601 int zmq::socket_base_t::send (msg_t *msg_, int flags_)
602 {
603  // Check whether the library haven't been shut down yet.
604  if (unlikely (ctx_terminated)) {
605  errno = ETERM;
606  return -1;
607  }
608 
609  // Check whether message passed to the function is valid.
610  if (unlikely (!msg_ || !msg_->check ())) {
611  errno = EFAULT;
612  return -1;
613  }
614 
615  // Process pending commands, if any.
616  int rc = process_commands (0, true);
617  if (unlikely (rc != 0))
618  return -1;
619 
620  // Clear any user-visible flags that are set on the message.
621  msg_->reset_flags (msg_t::more);
622 
623  // At this point we impose the flags on the message.
624  if (flags_ & ZMQ_SNDMORE)
625  msg_->set_flags (msg_t::more);
626 
627  // Try to send the message.
628  rc = xsend (msg_, flags_);
629  if (rc == 0)
630  return 0;
631  if (unlikely (errno != EAGAIN))
632  return -1;
633 
634  // In case of non-blocking send we'll simply propagate
635  // the error - including EAGAIN - up the stack.
636  if (flags_ & ZMQ_DONTWAIT || options.sndtimeo == 0)
637  return -1;
638 
639  // Compute the time when the timeout should occur.
640  // If the timeout is infite, don't care.
641  int timeout = options.sndtimeo;
642  uint64_t end = timeout < 0 ? 0 : (clock.now_ms () + timeout);
643 
644  // Oops, we couldn't send the message. Wait for the next
645  // command, process it and try to send the message again.
646  // If timeout is reached in the meantime, return EAGAIN.
647  while (true) {
648  if (unlikely (process_commands (timeout, false) != 0))
649  return -1;
650  rc = xsend (msg_, flags_);
651  if (rc == 0)
652  break;
653  if (unlikely (errno != EAGAIN))
654  return -1;
655  if (timeout > 0) {
656  timeout = (int) (end - clock.now_ms ());
657  if (timeout <= 0) {
658  errno = EAGAIN;
659  return -1;
660  }
661  }
662  }
663  return 0;
664 }

666 int zmq::socket_base_t::recv (msg_t *msg_, int flags_)
667 {
668  // Check whether the library haven't been shut down yet.
669  if (unlikely (ctx_terminated)) {
670  errno = ETERM;
671  return -1;
672  }
673 
674  // Check whether message passed to the function is valid.
675  if (unlikely (!msg_ || !msg_->check ())) {
676  errno = EFAULT;
677  return -1;
678  }
679 
680  // Get the message.
681  int rc = xrecv (msg_, flags_);
682  if (unlikely (rc != 0 && errno != EAGAIN))
683  return -1;
684 
685  // Once every inbound_poll_rate messages check for signals and process
686  // incoming commands. This happens only if we are not polling altogether
687  // because there are messages available all the time. If poll occurs,
688  // ticks is set to zero and thus we avoid this code.
689  //
690  // Note that 'recv' uses different command throttling algorithm (the one
691  // described above) from the one used by 'send'. This is because counting
692  // ticks is more efficient than doing RDTSC all the time.
693  if (++ticks == inbound_poll_rate) {
694  if (unlikely (process_commands (0, false) != 0))
695  return -1;
696  ticks = 0;
697  }
698 
699  // If we have the message, return immediately.
700  if (rc == 0) {
701  extract_flags (msg_);
702  return 0;
703  }
704 
705  // If the message cannot be fetched immediately, there are two scenarios.
706  // For non-blocking recv, commands are processed in case there's an
707  // activate_reader command already waiting int a command pipe.
708  // If it's not, return EAGAIN.
709  if (flags_ & ZMQ_DONTWAIT || options.rcvtimeo == 0) {
710  if (unlikely (process_commands (0, false) != 0))
711  return -1;
712  ticks = 0;
713 
714  rc = xrecv (msg_, flags_);
715  if (rc < 0)
716  return rc;
717  extract_flags (msg_);
718  return 0;
719  }
720 
721  // Compute the time when the timeout should occur.
722  // If the timeout is infite, don't care.
723  int timeout = options.rcvtimeo;
724  uint64_t end = timeout < 0 ? 0 : (clock.now_ms () + timeout);
725 
726  // In blocking scenario, commands are processed over and over again until
727  // we are able to fetch a message.
728  bool block = (ticks != 0);
729  while (true) {
730  if (unlikely (process_commands (block ? timeout : 0, false) != 0))
731  return -1;
732  rc = xrecv (msg_, flags_);
733  if (rc == 0) {
734  ticks = 0;
735  break;
736  }
737  if (unlikely (errno != EAGAIN))
738  return -1;
739  block = true;
740  if (timeout > 0) {
741  timeout = (int) (end - clock.now_ms ());
742  if (timeout <= 0) {
743  errno = EAGAIN;
744  return -1;
745  }
746  }
747  }
748 
749  extract_flags (msg_);
750  return 0;
751 }

789 int zmq::socket_base_t::process_commands (int timeout_, bool throttle_)
790 {
791  int rc;
792  command_t cmd;
793  if (timeout_ != 0) {
794 
795  // If we are asked to wait, simply ask mailbox to wait.
796  rc = mailbox.recv (&cmd, timeout_);
797  }
798  else {
799 
800  // If we are asked not to wait, check whether we haven't processed
801  // commands recently, so that we can throttle the new commands.
802 
803  // Get the CPU's tick counter. If 0, the counter is not available.
804  uint64_t tsc = zmq::clock_t::rdtsc ();
805 
806  // Optimised version of command processing - it doesn't have to check
807  // for incoming commands each time. It does so only if certain time
808  // elapsed since last command processing. Command delay varies
809  // depending on CPU speed: It's ~1ms on 3GHz CPU, ~2ms on 1.5GHz CPU
810  // etc. The optimisation makes sense only on platforms where getting
811  // a timestamp is a very cheap operation (tens of nanoseconds).
812  if (tsc && throttle_) {
813 
814  // Check whether TSC haven't jumped backwards (in case of migration
815  // between CPU cores) and whether certain time have elapsed since
816  // last command processing. If it didn't do nothing.
817  if (tsc >= last_tsc && tsc - last_tsc <= max_command_delay)
818  return 0;
819  last_tsc = tsc;
820  }
821 
822  // Check whether there are any commands pending for this thread.
823  rc = mailbox.recv (&cmd, 0);
824  }
825 
826  // Process all available commands.
827  while (rc == 0) {
828  cmd.destination->process_command (cmd);
829  rc = mailbox.recv (&cmd, 0);
830  }
831 
832  if (errno == EINTR)
833  return -1;
834 
835  zmq_assert (errno == EAGAIN);
836 
837  if (ctx_terminated) {
838  errno = ETERM;
839  return -1;
840  }
841 
842  return 0;
843 }

1003 int zmq::socket_base_t::monitor (const char *addr_, int events_)
1004 {
1005  int rc;
1006  if (unlikely (ctx_terminated)) {
1007  errno = ETERM;
1008  return -1;
1009  }
1010 
1011  // Support deregistering monitoring endpoints as well
1012  if (addr_ == NULL) {
1013  stop_monitor ();
1014  return 0;
1015  }
1016 
1017  // Parse addr_ string.
1018  std::string protocol;
1019  std::string address;
1020  rc = parse_uri (addr_, protocol, address);
1021  if (rc != 0)
1022  return -1;
1023 
1024  rc = check_protocol (protocol);
1025  if (rc != 0)
1026  return -1;
1027 
1028  // Event notification only supported over inproc://
1029  if (protocol != "inproc") {
1030  errno = EPROTONOSUPPORT;
1031  return -1;
1032  }
1033 
1034  // Register events to monitor
1035  monitor_events = events_;
1036 
1037  monitor_socket = zmq_socket( get_ctx (), ZMQ_PAIR);
1038  if (monitor_socket == NULL)
1039  return -1;
1040 
1041  // Never block context termination on pending event messages
1042  int linger = 0;
1043  rc = zmq_setsockopt (monitor_socket, ZMQ_LINGER, &linger, sizeof (linger));
1044  if (rc == -1)
1045  stop_monitor ();
1046

1048  rc = zmq_bind (monitor_socket, addr_);
1049  if (rc == -1)
1050  stop_monitor ();
1051  return rc;
1052 }
