37 void zmq::io_object_t::plug (io_thread_t *io_thread_)
38 {
39  zmq_assert (io_thread_);
40  zmq_assert (!poller);
41 
42  // Retrieve the poller from the thread we are running in.
43  poller = io_thread_->get_poller ();
44 }

46 void zmq::io_object_t::unplug ()
47 {
48  zmq_assert (poller);
49 
50  // Forget about old poller in preparation to be migrated
51  // to a different I/O thread.
52  poller = NULL;
53 }
