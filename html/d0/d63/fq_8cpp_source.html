47 void zmq::fq_t::terminated (pipe_t *pipe_)
48 {
49  const pipes_t::size_type index = pipes.index (pipe_);
50 
51  // Remove the pipe from the list; adjust number of active pipes
52  // accordingly.
53  if (index < active) {
54  active--;
55  pipes.swap (index, active);
56  if (current == active)
57  current = 0;
58  }
59  pipes.erase (pipe_);
60 }

74 int zmq::fq_t::recvpipe (msg_t *msg_, pipe_t **pipe_)
75 {
76  // Deallocate old content of the message.
77  int rc = msg_->close ();
78  errno_assert (rc == 0);
79 
80  // Round-robin over the pipes to get the next message.
81  while (active > 0) {
82 
83  // Try to fetch new message. If we've already read part of the message
84  // subsequent part should be immediately available.
85  bool fetched = pipes [current]->read (msg_);
86 
87  // Note that when message is not fetched, current pipe is deactivated
88  // and replaced by another active pipe. Thus we don't have to increase
89  // the 'current' pointer.
90  if (fetched) {
91  if (pipe_)
92  *pipe_ = pipes [current];
93  more = msg_->flags () & msg_t::more? true: false;
94  if (!more)
95  current = (current + 1) % active;
96  return 0;
97  }
98 
99  // Check the atomicity of the message.
100  // If we've already received the first part of the message
101  // we should get the remaining parts without blocking.
102  zmq_assert (!more);
103 
104  active--;
105  pipes.swap (current, active);
106  if (current == active)
107  current = 0;
108  }
109 
110  // No message is available. Initialise the output parameter
111  // to be a 0-byte message.
112  rc = msg_->init ();
113  errno_assert (rc == 0);
114  errno = EAGAIN;
115  return -1;
116 }

118 bool zmq::fq_t::has_in ()
119 {
120  // There are subsequent parts of the partly-read message available.
121  if (more)
122  return true;
123 
124  // Note that messing with current doesn't break the fairness of fair
125  // queueing algorithm. If there are no messages available current will
126  // get back to its original value. Otherwise it'll point to the first
127  // pipe holding messages, skipping only pipes with no messages available.
128  while (active > 0) {
129  if (pipes [current]->check_read ())
130  return true;
131 
132  // Deactivate the pipe.
133  active--;
134  pipes.swap (current, active);
135  if (current == active)
136  current = 0;
137  }
138 
139  return false;
140 }
